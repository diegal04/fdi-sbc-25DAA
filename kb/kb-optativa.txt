    Residuo plan reciclaje <- Residuo reciclable true .

        # C.1 Residuos en cantidades grandes

        # Solo se manda a la planta de reciclaje si se produce el residuo en grandes cantidades
        Residuo destino planta_reciclaje <- Residuo plan reciclaje, Residuo cantidad alta, empresa contrato planta_reciclaje . 

        # C.1 Residuos en cantidades pequeñas o medianas
        # Contenedores Urbanos / Recogida Selectiva

        # Se deshace de los residuos en contenedores si es en cantidades razonables o si la empresa no tiene contrato con la planta_reciclaje.
        Residuo destino Contenedor <- Residuo plan reciclaje, Residuo apto Contenedor, Residuo clasificado no_peligroso, Residuo cantidad baja .  
        Residuo destino Contenedor <- Residuo plan reciclaje, Residuo apto Contenedor, Residuo clasificado no_peligroso, Residuo cantidad media .

    # D) PLAN DE VALORIZACIÓN (Energética y Termoquímica)
    # REGLAS DE EXCLUSIÓN PARA VALORIZACIÓN

        # D.1 Valorización Energética (Biogás)
        
        Residuo aplicable valorizacion <- Residuo composicion materia_organica, Residuo cantidad alta .

        Residuo plan valorizacion <- Residuo aplicable valorizacion, Residuo minimizable false, Residuo reutilizable false, Residuo reciclable false, empresa prioridad minimizacion_impacto .

        Residuo destino planta_digestion_anaerobica <- Residuo plan valorizacion, Residuo composicion materia_organica, Residuo propiedad biodegradable, empresa contrato planta_anaerobica .

        # Beneficios para la empresa (Output de la planta)
        empresa genera biogas <- Residuo destino planta_digestion_anaerobica .
        empresa genera digestato <- Residuo destino planta_digestion_anaerobica .
        empresa capacidad cogeneracion <- empresa genera biogas .
        empresa reduce impacto_medioambiental <- empresa capacidad cogeneracion .

        # D.2 Valorización Termoquímica (Pirólisis)

        Residuo destino planta_de_pirolisis <- Residuo plan valorizacion, Residuo composicion materia_organica, Residuo propiedad no_biodegradable, empresa contrato planta_de_pirolisis .

        empresa genera gases_combustibles_val <- Residuo destino planta_de_pirolisis .
        empresa capacidad autoconsumo <- empresa genera gases_combustibles_val .
        empresa reduce impacto_medioambiental <- empresa capacidad autoconsumo .

        # D.3 Si el usuario indica que la empresa no tiene los contratos correspondientes

        Residuo destino vertedero_controlado <- Residuo aplicable valorizacion, Residuo minimizable false, Residuo reutilizable false, Residuo reciclable false, empresa prioridad reduccion_costos .
        Residuo destino vertedero_controlado <- Residuo plan valorizacion, empresa no_contrato planta_anaerobica, empresa no_contrato planta_de_pirolisis .

    # E) PLANES ESPECIALES
    Residuo plan tratamiento_especial <- Residuo clasificado peligroso, Residuo reciclable true .
    Residuo destino punto_limpio_municipal <- Residuo plan tratamiento_especial, Residuo apto punto_limpio .

    Residuo plan desmontaje_especial <- Residuo clasificado raee, Residuo reutilizable false .
    Residuo destino planta_tratamiento_raee <- Residuo plan desmontaje_especial, empresa contrato planta_raee .
    Residuo destino vertedero_controlado <- Residuo plan desmontaje_especial, empresa no_contrato planta_raee .

    # Excepción por volumen
    Residuo destino punto_limpio_municipal <- Residuo composicion madera, Residuo cantidad alta .

    # F) PLAN DE ELIMINACIÓN (Vertedero, ultima opción)
    Residuo plan eliminacion <- Residuo clasificado no_peligroso, Residuo reutilizable false, Residuo minimizable false, Residuo reciclable false .
    Residuo destino vertedero_controlado <- Residuo plan eliminacion, Residuo cantidad alta .

    Residuo destino contenedor_gris <- Residuo plan eliminacion, Residuo clasificado no_peligroso, Residuo cantidad baja .
